234 template<typename E>
235 void EdgeExporter<E>::edgeRemoved(boost::shared_ptr<E> edge, std::map<int, std::vector<std::pair<AgentId, AgentId> > >& removeMap) {
236  for (EdgeMapIterator emIter = exportedEdges.begin(); emIter != exportedEdges.end(); ++emIter) {
237  std::vector<boost::shared_ptr<E> >* edges = emIter->second;
238  for (typename std::vector<boost::shared_ptr<E> >::iterator edgeIter = edges->begin(); edgeIter != edges->end();) {
239  if (*edgeIter == edge) {
240  std::map<int, std::vector<std::pair<AgentId, AgentId> > >::iterator iter =
241  removeMap.find(emIter->first);
242  if (iter == removeMap.end()) {
243  std::vector<std::pair<AgentId, AgentId> > vec;
244  vec.push_back(std::make_pair<AgentId, AgentId>(edge->source()->getId(), edge->target()->getId()));
245  removeMap[emIter->first] = vec;
246  } else {
247  iter->second.push_back(std::make_pair<AgentId, AgentId>(edge->source()->getId(),
248  edge->target()->getId()));
249  }
250  edgeIter = edges->erase(edgeIter);
251  } else {
252  ++edgeIter;
253  }
254  }
255  }

258 template<typename E>
259 void EdgeExporter<E>::cleanUp() {
260  // just delete the edge lists themselves. We don't delete the edges as they are
261  // part of the network itself.
262  for (EdgeMapIterator emIter = edgesToExport.begin(); emIter != edgesToExport.end(); ++emIter) {
263  std::vector<boost::shared_ptr<E> >* edges = emIter->second;
264  int exportTo = emIter->first;
265  if (exportedEdges.find(exportTo) == exportedEdges.end()) {
266  std::vector<boost::shared_ptr<E> >* exportedVec = new std::vector<boost::shared_ptr<E> >();
267  exportedEdges[exportTo] = exportedVec;
268  exportedVec->insert(exportedVec->end(), edges->begin(), edges->end());
269 
270  } else {
271  std::vector<boost::shared_ptr<E> >* exportedVec = exportedEdges[exportTo];
272  exportedVec->insert(exportedVec->end(), edges->begin(), edges->end());
273 
274  }
275  delete edges;
276  }
277  edgesToExport.clear();
278 }

470 template<typename E, typename EdgeContent, typename ProviderReceiver>
471 void sendContent(EdgeExporter<E>& exporter, std::vector<boost::mpi::request>& requests, ProviderReceiver& provider, boost::ptr_vector<std::vector<EdgeContent> > &sendBuffer) {
472  std::map<int, std::vector<boost::shared_ptr<E> >*>& edgesToExport = exporter.getEdgesToExport();
473  boost::mpi::communicator* comm = RepastProcess::instance()->getCommunicator();
474 
475  std::vector<EdgeContent>* edgeContent;
476 
477  for (typename EdgeExporter<E>::EdgeMapIterator emIter = edgesToExport.begin(); emIter != edgesToExport.end(); ++emIter) {
478  // the process to send the edge to
479  int receiver = emIter->first;
480  std::vector<boost::shared_ptr<E> >* edges = emIter->second;
481  sendBuffer.push_back(edgeContent = new std::vector<EdgeContent>);
482  // check end points and add any local nodes to the AgentExporter
483  for (typename std::vector<boost::shared_ptr<E> >::iterator iter = edges->begin(); iter != edges->end(); ++iter) {
484 
485  boost::shared_ptr<E> edge = *iter;
486  AgentId srcId = edge->source()->getId();
487  AgentId targetId = edge->target()->getId();
488  if (srcId.currentRank() == comm->rank()) {
489  RepastProcess::instance()->addExportedAgent(receiver, srcId);
490  } else if (srcId.currentRank() != receiver) {
491  // sending node that's foreign to both this P and the receiver
492  // so need to tell the P where the node resides to export it to
493  // the receiver.
494  exporter.addAgentExportRequest(receiver, srcId);
495  }
496 
497  if (targetId.currentRank() == comm->rank()) {
498  RepastProcess::instance()->addExportedAgent(receiver, targetId);
499  } else if (targetId.currentRank() != receiver) {
500  // sending node that's foreign to both this P and the receiver
501  // so need to tell the P where the node resides to export it to
502  // the receiver.
503  exporter.addAgentExportRequest(receiver, targetId);
504  }
505 
506  provider.provideEdgeContent(edge, *edgeContent);
507  }
508  requests.push_back(comm->isend(receiver, NET_EDGE_UPDATE, *edgeContent)); //*edges));
509  }
510 }

631 /**
632  * Synchronizes any edges that have been created as complementary edges. This only
633  * necessary if the edges have been deleted or their state has been changed in some way.
634  *
635  * @param net the network in which to create the complementary edges or from which to send
636  * complementary edges
637  * @param edgeManager updates edges from EdgeContent and creates EdgeContent from an edge and a context.
638  *
639  * @tparam Vertex the vertex (agent) type
640  * @tparam Edge the edge type
641  * @tparam EdgeContent the serializable struct or class that describes edge state.
642  * @tparam EdgeManager updates edges from EdgeContent and provides EdgeContent given a context and an edge of type Edge. It must
643  * implement void provideEdgeContent(const Edge* edge, std::vector<EdgeContent>& edgeContent) and
644  * void receiveEdgeContent(const EdgeContent& content);
645  */
646 template<typename Vertex, typename Edge, typename EdgeContent, typename EdgeManager>
647 void synchEdges(SharedNetwork<Vertex, Edge, EdgeContent, EdgeManager>* net, EdgeManager& edgeManager) {
648 
649  std::map<int, std::vector<boost::shared_ptr<Edge> >* >& exports = net->edgeExporter.getExportedEdges();
650  boost::mpi::communicator* world = RepastProcess::instance()->getCommunicator();
651 
652  std::vector<boost::mpi::request> requests;
653  std::vector<std::vector<EdgeContent>*> contents;
654 
655  // receive edge content from those this P is importing from
656  for (std::map<int, int>::const_iterator iter = net->senders.begin(); iter != net->senders.end(); ++iter) {
657  int sender = iter->first;
658  std::vector<EdgeContent>* content = new std::vector<EdgeContent>();
659  requests.push_back(world->irecv(sender, NET_EDGE_SYNC, *content));
660  contents.push_back(content);
661  }
662 
663  boost::ptr_vector<std::vector<EdgeContent> >* contentVector = new boost::ptr_vector<std::vector<EdgeContent> >;
664  std::vector<EdgeContent>* edgeContent;
665 
666  // send updated edge content to those I'm exporting to
667  for (typename EdgeExporter<Edge>::EdgeMapIterator emIter = exports.begin(); emIter != exports.end(); ++emIter) {
668  // the process to send the edge to
669  int receiver = emIter->first;
670  std::vector<boost::shared_ptr<Edge> >* edges = emIter->second;
671  contentVector->push_back( edgeContent = new std::vector<EdgeContent>);
672  for (typename std::vector<boost::shared_ptr<Edge> >::iterator iter = edges->begin(); iter != edges->end(); ++iter) {
673  edgeContent->push_back(edgeManager.provideEdgeContent(iter->get()));
674  }
675  requests.push_back(world->isend(receiver, NET_EDGE_SYNC, *edgeContent));
676  }
677 
678  boost::mpi::wait_all(requests.begin(), requests.end());
679  delete contentVector;
680 
681  for (int i = 0, n = contents.size(); i < n; i++) {
682  std::vector<EdgeContent>* content = contents[i];
683  for (typename std::vector<EdgeContent>::const_iterator iter = content->begin(); iter != content->end(); ++iter) {
684  edgeManager.receiveEdgeContent(*iter);
685  }
686  delete content;
687  }
688 }
